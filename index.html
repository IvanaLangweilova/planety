function planety() close all; clc; %------------------------------------------------------------ % ZAKLADNI NASTAVENI % jednotky: % vzdalenost = AU % cas = roky % hmotnost = nasobky hmotnosti Slunce %------------------------------------------------------------ G = 4*pi^2; fig = figure( ... 'Name', 'Planetarni soustava', ... 'NumberTitle', 'off', ... 'Color', [0.05 0.05 0.08], ... 'Position', [80 60 1350 780]); ax = axes( ... 'Parent', fig, ... 'Position', [0.05 0.08 0.66 0.84], ... 'Color', [0.03 0.03 0.06], ... 'XColor', 'w', ... 'YColor', 'w', ... 'GridColor', [0.35 0.35 0.45], ... 'FontSize', 10); hold(ax, 'on'); axis(ax, 'equal'); axis(ax, [-15 15 -15 15]); grid(ax, 'on'); xlabel(ax, 'x [AU]', 'Color', 'w'); ylabel(ax, 'y [AU]', 'Color', 'w'); title(ax, 'Planetarni soustava', 'Color', 'w', 'FontSize', 16); % Hvezdne pozadi rand('seed', 1); nStars = 260; sx = -15 + 30*rand(nStars,1); sy = -15 + 30*rand(nStars,1); ss = 4 + 16*rand(nStars,1); scatter(ax, sx, sy, ss, 'w', 'filled', ... 'MarkerFaceAlpha', 0.6, 'MarkerEdgeAlpha', 0.6); %------------------------------------------------------------ % OVLADACI PANEL %------------------------------------------------------------ panel = uipanel( ... 'Parent', fig, ... 'Title', 'Ovládání simulace', ... 'FontSize', 11, ... 'ForegroundColor', 'w', ... 'BackgroundColor', [0.10 0.10 0.14], ... 'Position', [0.75 0.08 0.22 0.84]); uicontrol(panel, ... 'Style', 'text', ... 'Units', 'normalized', ... 'Position', [0.08 0.90 0.84 0.05], ... 'String', 'Posun posuvníku okamžitě změní dráhy.', ... 'BackgroundColor', [0.10 0.10 0.14], ... 'ForegroundColor', 'w', ... 'FontSize', 10, ... 'HorizontalAlignment', 'left'); txtEarth = uicontrol(panel, ... 'Style', 'text', ... 'Units', 'normalized', ... 'Position', [0.08 0.82 0.84 0.05], ... 'String', 'Rychlost Země: 1.00x', ... 'BackgroundColor', [0.10 0.10 0.14], ... 'ForegroundColor', [0.4 0.8 1.0], ... 'FontSize', 12, ... 'FontWeight', 'bold'); sEarth = uicontrol(panel, ... 'Style', 'slider', ... 'Units', 'normalized', ... 'Position', [0.08 0.78 0.84 0.04], ... 'Min', 0.70, 'Max', 1.30, 'Value', 1.00, ... 'Callback', @(src,evt) slider_changed(src, fig, 'earth')); txtSun = uicontrol(panel, ... 'Style', 'text', ... 'Units', 'normalized', ... 'Position', [0.08 0.68 0.84 0.05], ... 'String', 'Hmotnost Slunce: 1.00x', ... 'BackgroundColor', [0.10 0.10 0.14], ... 'ForegroundColor', [1.0 0.9 0.4], ... 'FontSize', 12, ... 'FontWeight', 'bold'); sSun = uicontrol(panel, ... 'Style', 'slider', ... 'Units', 'normalized', ... 'Position', [0.08 0.64 0.84 0.04], ... 'Min', 0.70, 'Max', 1.50, 'Value', 1.00, ... 'Callback', @(src,evt) slider_changed(src, fig, 'sun')); txtCometV = uicontrol(panel, ... 'Style', 'text', ... 'Units', 'normalized', ... 'Position', [0.08 0.54 0.84 0.05], ... 'String', 'Rychlost komety: 5.50 AU/rok', ... 'BackgroundColor', [0.10 0.10 0.14], ... 'ForegroundColor', [0.6 1.0 1.0], ... 'FontSize', 12, ... 'FontWeight', 'bold'); sCometV = uicontrol(panel, ... 'Style', 'slider', ... 'Units', 'normalized', ... 'Position', [0.08 0.50 0.84 0.04], ... 'Min', 3.0, 'Max', 8.0, 'Value', 5.5, ... 'Callback', @(src,evt) slider_changed(src, fig, 'cometv')); txtCometA = uicontrol(panel, ... 'Style', 'text', ... 'Units', 'normalized', ... 'Position', [0.08 0.40 0.84 0.05], ... 'String', 'Úhel komety: -8.0°', ... 'BackgroundColor', [0.10 0.10 0.14], ... 'ForegroundColor', [0.5 0.9 1.0], ... 'FontSize', 12, ... 'FontWeight', 'bold'); sCometA = uicontrol(panel, ... 'Style', 'slider', ... 'Units', 'normalized', ... 'Position', [0.08 0.36 0.84 0.04], ... 'Min', -35, 'Max', 25, 'Value', -8, ... 'Callback', @(src,evt) slider_changed(src, fig, 'cometa')); btnPause = uicontrol(panel, ... 'Style', 'togglebutton', ... 'Units', 'normalized', ... 'Position', [0.08 0.22 0.38 0.07], ... 'String', 'Pauza', ... 'FontSize', 11, ... 'Value', 0, ... 'Callback', @(src,evt) toggle_pause(src, fig)); btnReset = uicontrol(panel, ... 'Style', 'pushbutton', ... 'Units', 'normalized', ... 'Position', [0.54 0.22 0.38 0.07], ... 'String', 'Reset', ... 'FontSize', 11, ... 'Callback', @(src,evt) reset_sim(fig)); uicontrol(panel, ... 'Style', 'text', ... 'Units', 'normalized', ... 'Position', [0.08 0.03 0.84 0.14], ... 'String', ['Didaktický tip:' sprintf('\n') ... '- menší rychlost Země -> menší dráha' sprintf('\n') ... '- větší rychlost Země -> protáhlá elipsa' sprintf('\n') ... '- změna komety -> průlet nebo gravitační prak'], ... 'BackgroundColor', [0.10 0.10 0.14], ... 'ForegroundColor', 'w', ... 'FontSize', 10, ... 'HorizontalAlignment', 'left'); %------------------------------------------------------------ % STRUKTURA APLIKACE %------------------------------------------------------------ app = struct(); app.fig = fig; app.ax = ax; app.G = G; app.dt = 0.0015; app.time = 0; app.trail_len = 1400; % Měsíc je ve skutečnosti velmi blízko Země. % Tato hodnota zvětší pouze jeho ZOBRAZENÍ, ne fyzikální výpočet. app.moonVisualScale = 80; app.changed = true; app.paused = false; app.sEarth = sEarth; app.sSun = sSun; app.sCometV = sCometV; app.sCometA = sCometA; app.txtEarth = txtEarth; app.txtSun = txtSun; app.txtCometV = txtCometV; app.txtCometA = txtCometA; app.btnPause = btnPause; app.sunPlot = []; app.bodyPlots = []; app.trailPlots = []; app.textPlots = []; app.ringPlot = []; app.moonOrbitPlot = []; app.bodies = []; guidata(fig, app); %------------------------------------------------------------ % HLAVNI SMYCKA %------------------------------------------------------------ while ishandle(fig) app = guidata(fig); if app.changed app = initialize_simulation(app); guidata(fig, app); end if ~app.paused app = step_simulation(app); app = redraw_scene(app); guidata(fig, app); end drawnow(); pause(0.02); end end %============================================================== % CALLBACKY %============================================================== function slider_changed(src, fig, what) app = guidata(fig); val = get(src, 'Value'); switch what case 'earth' set(app.txtEarth, 'String', sprintf('Rychlost Země: %.2fx', val)); case 'sun' set(app.txtSun, 'String', sprintf('Hmotnost Slunce: %.2fx', val)); case 'cometv' set(app.txtCometV, 'String', sprintf('Rychlost komety: %.2f AU/rok', val)); case 'cometa' set(app.txtCometA, 'String', sprintf('Úhel komety: %.1f°', val)); end app.changed = true; guidata(fig, app); end function toggle_pause(src, fig) app = guidata(fig); app.paused = logical(get(src, 'Value')); if app.paused set(src, 'String', 'Pokračovat'); else set(src, 'String', 'Pauza'); end guidata(fig, app); end function reset_sim(fig) app = guidata(fig); app.time = 0; app.changed = true; app.paused = false; set(app.btnPause, 'Value', 0, 'String', 'Pauza'); guidata(fig, app); end %============================================================== % INICIALIZACE SIMULACE %============================================================== function app = initialize_simulation(app) if ~isempty(app.sunPlot) && ishandle(app.sunPlot) delete(app.sunPlot); end for k = 1:numel(app.bodyPlots) if ishandle(app.bodyPlots(k)) delete(app.bodyPlots(k)); end end for k = 1:numel(app.trailPlots) if ishandle(app.trailPlots(k)) delete(app.trailPlots(k)); end end for k = 1:numel(app.textPlots) if ishandle(app.textPlots(k)) delete(app.textPlots(k)); end end if ~isempty(app.ringPlot) && ishandle(app.ringPlot) delete(app.ringPlot); end if isfield(app, 'moonOrbitPlot') && ~isempty(app.moonOrbitPlot) && ishandle(app.moonOrbitPlot) delete(app.moonOrbitPlot); end app.time = 0; G = app.G; Msun = get(app.sSun, 'Value'); earth_factor = get(app.sEarth, 'Value'); comet_speed = get(app.sCometV, 'Value'); comet_angle = deg2rad(get(app.sCometA, 'Value')); % Hmotnosti M_earth = 3.0e-6; M_saturn = 2.86e-4; M_moon = 3.7e-8; M_comet = 1.0e-10; %------------------------------------------------------------ % Telesa % 1 = Země % 2 = Saturn % 3 = Měsíc % 4 = Kometa %------------------------------------------------------------ bodies = struct(); % Země rE = 1.0; vE = earth_factor * sqrt(G * Msun / rE); bodies(1).name = 'Země'; bodies(1).mass = M_earth; bodies(1).color = [0.2 0.7 1.0]; bodies(1).size = 8; bodies(1).pos = [rE, 0]; bodies(1).vel = [0, vE]; bodies(1).trailx = bodies(1).pos(1); bodies(1).traily = bodies(1).pos(2); % Saturn rS = 9.5; vS = sqrt(G * Msun / rS); bodies(2).name = 'Saturn'; bodies(2).mass = M_saturn; bodies(2).color = [0.95 0.80 0.55]; bodies(2).size = 12; bodies(2).pos = [rS, 0]; bodies(2).vel = [0, 0.95*vS]; bodies(2).trailx = bodies(2).pos(1); bodies(2).traily = bodies(2).pos(2); % Měsíc kolem Země % Skutečná vzdálenost Měsíce od Země je přibližně 0.00257 AU. % V grafu by to téměř nebylo vidět, proto se později pouze zobrazení zvětšuje. moon_r = 0.00257; moon_v_rel = sqrt(G * M_earth / moon_r); earth_pos = bodies(1).pos; earth_vel = bodies(1).vel; bodies(3).name = 'Měsíc'; bodies(3).mass = M_moon; bodies(3).color = [0.88 0.88 0.88]; bodies(3).size = 5; bodies(3).pos = earth_pos + [moon_r, 0]; bodies(3).vel = earth_vel + [0, moon_v_rel]; bodies(3).trailx = bodies(3).pos(1); bodies(3).traily = bodies(3).pos(2); % Kometa bodies(4).name = 'Kometa'; bodies(4).mass = M_comet; bodies(4).color = [0.6 1.0 1.0]; bodies(4).size = 6; bodies(4).pos = [-13, 3.0]; bodies(4).vel = comet_speed * [cos(comet_angle), sin(comet_angle)]; bodies(4).trailx = bodies(4).pos(1); bodies(4).traily = bodies(4).pos(2); app.bodies = bodies; app.changed = false; %------------------------------------------------------------ % GRAFICKE OBJEKTY %------------------------------------------------------------ app.sunPlot = plot(app.ax, 0, 0, 'o', ... 'MarkerSize', 18, ... 'MarkerFaceColor', [1 0.9 0.1], ... 'MarkerEdgeColor', [1 0.95 0.4]); n = numel(app.bodies); app.trailPlots = gobjects(1,n); app.bodyPlots = gobjects(1,n); app.textPlots = gobjects(1,n); for i = 1:n if i == 3 % Zobrazení Měsíce je graficky zvětšené vůči Zemi. [px, py, tx, ty] = moon_display_data(app); else px = app.bodies(i).pos(1); py = app.bodies(i).pos(2); tx = app.bodies(i).trailx; ty = app.bodies(i).traily; end app.trailPlots(i) = plot(app.ax, tx, ty, '-', ... 'Color', app.bodies(i).color * 0.8, ... 'LineWidth', 1.2); app.bodyPlots(i) = plot(app.ax, px, py, 'o', ... 'MarkerSize', app.bodies(i).size, ... 'MarkerFaceColor', app.bodies(i).color, ... 'MarkerEdgeColor', app.bodies(i).color); app.textPlots(i) = text(app.ax, px+0.12, py+0.12, ... app.bodies(i).name, ... 'Color', 'w', ... 'FontSize', 9); end % Saturnuv prstenec th = linspace(0, 2*pi, 200); sat = app.bodies(2).pos; ring_x = sat(1) + 0.85*cos(th); ring_y = sat(2) + 0.28*sin(th); app.ringPlot = plot(app.ax, ring_x, ring_y, '-', ... 'Color', [0.9 0.9 0.8], ... 'LineWidth', 1.0); % Zvýrazněná kruhová dráha Měsíce kolem Země earth = app.bodies(1).pos; moon_r_visual = 0.00257 * app.moonVisualScale; moon_orbit_x = earth(1) + moon_r_visual * cos(th); moon_orbit_y = earth(2) + moon_r_visual * sin(th); app.moonOrbitPlot = plot(app.ax, moon_orbit_x, moon_orbit_y, '--', ... 'Color', [0.7 0.7 0.7], ... 'LineWidth', 0.8); title(app.ax, 'Planetární soustava | čas = 0.00 roku', ... 'Color', 'w', ... 'FontSize', 16); end %============================================================== % JEDEN KROK SIMULACE %============================================================== function app = step_simulation(app) dt = app.dt; n = numel(app.bodies); pos = zeros(n,2); vel = zeros(n,2); mass = zeros(n,1); Msun = get(app.sSun, 'Value'); for i = 1:n pos(i,:) = app.bodies(i).pos; vel(i,:) = app.bodies(i).vel; mass(i) = app.bodies(i).mass; end acc = compute_acc(pos, mass, Msun, app.G); % Semi-implicit Euler vel = vel + acc * dt; pos = pos + vel * dt; for i = 1:n app.bodies(i).pos = pos(i,:); app.bodies(i).vel = vel(i,:); app.bodies(i).trailx(end+1) = pos(i,1); app.bodies(i).traily(end+1) = pos(i,2); if numel(app.bodies(i).trailx) > app.trail_len app.bodies(i).trailx = app.bodies(i).trailx(end-app.trail_len+1:end); app.bodies(i).traily = app.bodies(i).traily(end-app.trail_len+1:end); end end app.time = app.time + dt; end %============================================================== % VYPOCET ZRYCHLENI %============================================================== function acc = compute_acc(pos, mass, Msun, G) n = size(pos,1); acc = zeros(n,2); % Změkčení výpočtu, aby při velmi malých vzdálenostech nevznikaly nekonečné síly. eps2 = 1e-8; for i = 1:n % Gravitační vliv pevného Slunce v počátku r = -pos(i,:); d2 = r(1)^2 + r(2)^2 + eps2; d = sqrt(d2); acc(i,:) = acc(i,:) + G * Msun * r / (d^3); % Vzájemná gravitace těles for j = 1:n if j ~= i rij = pos(j,:) - pos(i,:); d2 = rij(1)^2 + rij(2)^2 + eps2; d = sqrt(d2); acc(i,:) = acc(i,:) + G * mass(j) * rij / (d^3); end end end end %============================================================== % PREKRESLENI %============================================================== function app = redraw_scene(app) n = numel(app.bodies); for i = 1:n if i == 3 % Měsíc se fyzikálně počítá správně, % ale jeho zobrazení vůči Zemi je zvětšené. [px, py, tx, ty] = moon_display_data(app); else px = app.bodies(i).pos(1); py = app.bodies(i).pos(2); tx = app.bodies(i).trailx; ty = app.bodies(i).traily; end set(app.trailPlots(i), ... 'XData', tx, ... 'YData', ty); set(app.bodyPlots(i), ... 'XData', px, ... 'YData', py); set(app.textPlots(i), ... 'Position', [px+0.12, py+0.12, 0]); end % Aktualizace Saturnova prstence th = linspace(0, 2*pi, 200); sat = app.bodies(2).pos; ring_x = sat(1) + 0.85*cos(th); ring_y = sat(2) + 0.28*sin(th); set(app.ringPlot, ... 'XData', ring_x, ... 'YData', ring_y); % Aktualizace zvýrazněné dráhy Měsíce kolem Země earth = app.bodies(1).pos; moon_r_visual = 0.00257 * app.moonVisualScale; moon_orbit_x = earth(1) + moon_r_visual * cos(th); moon_orbit_y = earth(2) + moon_r_visual * sin(th); if isfield(app, 'moonOrbitPlot') && ishandle(app.moonOrbitPlot) set(app.moonOrbitPlot, ... 'XData', moon_orbit_x, ... 'YData', moon_orbit_y); end title(app.ax, sprintf('Planetární soustava | čas = %.2f roku', app.time), ... 'Color', 'w', ... 'FontSize', 16); end %============================================================== % POMOCNA FUNKCE PRO ZOBRAZENI MESICE %============================================================== function [px, py, tx, ty] = moon_display_data(app) % Fyzikální poloha Měsíce zůstává skutečná. % Zvětšuje se pouze zobrazení vůči poloze Země, % aby bylo vidět, že Měsíc obíhá kolem Země. scale = app.moonVisualScale; earth = app.bodies(1); moon = app.bodies(3); % Aktuální zvětšená poloha Měsíce vůči Zemi rel = moon.pos - earth.pos; display_pos = earth.pos + scale * rel; px = display_pos(1); py = display_pos(2); % Zvětšená stopa Měsíce vůči stopě Země L = min(numel(earth.trailx), numel(moon.trailx)); ex = earth.trailx(end-L+1:end); ey = earth.traily(end-L+1:end); mx = moon.trailx(end-L+1:end); my = moon.traily(end-L+1:end); tx = ex + scale * (mx - ex); ty = ey + scale * (my - ey); end
